		int sum =0;
		for(int i=1; i<=max; ++i) {          
			sum += i;                          //max의값만큼 숫자를더함
		}
		return sum;
	}	
	public static void main(String[] args) {

		System.out.println(result);             //1~10까지의 숫자를 합

		//조건 1 starNum이 endNum보다 크면 return 0

		public static int getConunt(int starNum, int endNum) {
			int count = 0;
			for(int i=starNum; i<=endNum;++i) {
				if(i%2 ==0) {
					++count;
				}
			}
			if(starNum >= endNum) {
				return 0;
			}if(endNum > 100) {
				return 0;
			}
			
			return count;
		}
