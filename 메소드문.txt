    문제예시 (max의 값만큼 숫자를 합하라)
    public static int getSum(int max) {
		int sum =0;
		for(int i=1; i<=max; ++i) {          
			sum += i;                          //max의값만큼 숫자를더함
		}
		return sum;
	}	
	public static void main(String[] args) {
		int result = getSum(10);                //int max에 10의숫자를 넣줌
		System.out.println(result);             //1~10까지의 숫자를 합

	}
