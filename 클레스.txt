//피자라는 객체(object)생성
public class Pizza {	
	public void 도우를만들다() {
		System.out.println("도우 만드는 중~");
	}
	public void 피자를굽다() {
		System.out.println("피자 굽는 중~");
	}
	public void 토핑을추가하다(String 토핑) {
		System.out.println(토핑 + " 토핑 추가!");
	}
	private void 비법소스를첨가하다() {                                    //private는 해당 메소드에서만 출력
		System.out.println("최고급 트러플 오일 듬뿍!");
	}

	
}//캡슐화

public class Client {

	public static void main(String[] args) {
		//피자 객체(object) 호출
		int x = 10;
		Pizza 하와이안피자_치즈둠뿍 = new Pizza();                           //하와이안 피자 제작 과정	//Pizza를 객체(==인스턴스)화 했다. 위의 Pizza클레스를 불러옴	
		하와이안피자_치즈둠뿍.도우를만들다();                                 //도우 만드는 중~ 출력
		하와이안피자_치즈둠뿍.토핑을추가하다("버섯");                          //버섯 토핑 추가! 출력
		하와이안피자_치즈둠뿍.피자를굽다();                                   //피자 굽는 중~ 출력
		//불고기피자 만드는 과정
		Pizza 불고기피자 = new Pizza();                                     //Pizza 클래스 호출(불고기 피자 만드는 과정)
		불고기피자.도우를만들다();                                           //도우 만드는 중~ 출력
		불고기피자.토핑을추가하다("소고기");                                  //소고기 토핑 추가! 출력
		불고기피자.피자를굽다();                                             //피자 굽는 중~ 출력
		불고기피자.비법소스를첨가하다();                                      //private 이기떄문에 출력x

	}

}

public class Hamburger {
	//1. 햄버거 클래스 생성
	//*** 아래 메소드 생성 ***
  //1-1. public void 빵굽다()
	//System.out.println("빵굽는중 ~ ")
  //1-2. public String 패티추가하다 (String 패티이름)
  //1-3. public int 주문수량(int 주문수량)
	public  void 빵굽다() {
		System.out.println("빵굽는중 ~ ");
	}
	public  String 패티추가하다(String 패티) {
		return 패티;	                                  //String 패티 값에 패티를 리턴
	}
	public  int 주문수량(int 주문수량) {
		return 주문수량;                                  //int 주문수량에 주문수량을 리턴
	}
	
	
}

public class 고객 {

	public static void main(String[] args) {
		//Pizza 하와이안피자_치즈둠뿍 = new Pizza();
		Hamburger 불고기버거 = new Hamburger();
		불고기버거.빵굽다();
		String 패티 = 불고기버거.패티추가하다("불고기");       //리턴값 패티 =불고기 대입 "불고기패티추가" 출력
		System.out.println(패티 + " 패티 추가 ");
		int 주문수량 = 불고기버거.주문수량(5);                //리턴값 주문수랭 =5 대입 "5개 주문" 출력
		System.out.println(주문수량 + " 개 주문");
		

	}
}
//class는 객체(== Object)
public class School {
	int grade; //필드 변수(최강 전역 변수): 가장 오래 삶
	
	public void 등교하다() {
		System.out.println("학학학학 학교를 갔어!");
	}

    public void 수업듣다() {
		System.out.println("학학학학 학교에 수업을 들었어!");
	}
    
    public void 수업시간에자다() {
 		System.out.println("학학학학 학교에서 잤어!");
 	}
}

package 메소드_3일차;

public class 클래스호출 {
	
	public static School getSchool(School sch) {
		sch.grade = 20;
		sch.등교하다();
		sch.수업듣다();
		sch.수업시간에자다();
		return sch;
	}

	public static void main(String[] args) {
		School s = new School();//파일을 불러오기!		
		s = getSchool(s);
		System.out.println(s.grade);

	}

}
